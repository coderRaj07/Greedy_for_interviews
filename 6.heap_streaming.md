# 0️⃣ Median in a given stream of numbers (Amazon Favourite)

🔗 [https://leetcode.com/problems/find-median-from-data-stream/](https://leetcode.com/problems/find-median-from-data-stream/)

---

### Python

```python
class MedianFinder:
    # left can be (at most) one more than right
    # len(left) == len(right)+1 at max len of length can be len of right + 1 (len left can't exceed len right + 1)
    # and len(left) == len(right) at max len of right can be len of left (len right can't exceed len left)
    # if len(left)>len(right)+1 invalid
    # if len(right)>len(left) invalid as we want left to have one more element to carry
    def __init__(self):
        self.heap_l = []
        self.heap_r = []
        heapq.heapify(self.heap_l) # max heap
        heapq.heapify(self.heap_r) # min heap

    def addNum(self, num: int) -> None:
        #1. add to the left
        heapq.heappush(self.heap_l,(-num))

        #2. ordering
        if self.heap_r and self.heap_r[0]<-self.heap_l[0]:
            # pop from left and push to right
            heapq.heappush(self.heap_r,(-heapq.heappop(self.heap_l)))

        # resize
        if len(self.heap_l)>len(self.heap_r)+1:
            # pop from left and push to right
            heapq.heappush(self.heap_r,(-heapq.heappop(self.heap_l)))
        if len(self.heap_r)>len(self.heap_l):
            # pop from right and push to left
            heapq.heappush(self.heap_l,(-heapq.heappop(self.heap_r)))
        

    def findMedian(self) -> float:
        size = len(self.heap_r)+len(self.heap_l)
        if size%2==0:
            return (self.heap_r[0]+(-self.heap_l[0]))/2
        else:
            return -self.heap_l[0]
```

# 1️⃣ Reorganize String (Amazon Favourite)

🔗 [https://leetcode.com/problems/reorganize-string/](https://leetcode.com/problems/reorganize-string/)

### Problem

```
s = "aab"
```

Output

```
aba
```

No **two adjacent characters should be same**.

---

# Key Idea

Always place **most frequent character first**.

But if we place it immediately again → invalid.

So we:

```
1 take most frequent
2 block it temporarily
3 use second most frequent
```

---

### Visual

```
a:2
b:1
```

Heap:

```
a a b
```

Process:

```
take a
take b
take a
```

---

### Python

```python
class Solution:
    def reorganizeString(self, s: str) -> str:
        # PEPCODING
        # Blocking condition: if used now, keep in block position
        # Frequency condition: keep using max freq values
        obj= defaultdict(int)
        for i in range(len(s)):
            obj[s[i]]+=1

        heap=[]
        heapq.heapify(heap)
        
        for ch, v in obj.items():
            # max heap since max freq reqd. continuously
            heapq.heappush(heap, (-v, ch))

        res = []
        prev = (0, '') # freq, character
        while heap:
            count, ch = heapq.heappop(heap)
            # decrease count
            count +=1
            # add to result
            res.append(ch)
            # before marking this ch as prev
            # unblock and push the current prev value to heap
            if prev[0] < 0: # if prev has count which is negative, means freq exists
              heapq.heappush(heap, (prev[0], prev[1]))
            # mark ch as prev
            prev = (count, ch)

        final = "".join(res)

        if len(s)!=len(final):
            return ""

        return final
```

# 2️⃣ Minimum Cost to Connect Ropes

🔗 [https://www.geeksforgeeks.org/problems/minimum-cost-of-ropes](https://www.geeksforgeeks.org/problems/minimum-cost-of-ropes-1587115620/1)

### Problem

```
ropes = [4,3,2,6]
```

Join ropes → cost = sum of lengths.

Goal → **minimum cost**.

---

### Greedy Insight

Always connect **two smallest ropes first**.

Because large ropes increase future costs.

---

### Visual

```
2 + 3 = 5
heap → [4,5,6]

4 + 5 = 9
heap → [6,9]

6 + 9 = 15
```

Total

```
5 + 9 + 15
```

---

### Python

```python
import heapq

def minCost(arr):

    heapq.heapify(arr)

    cost=0

    while len(arr)>1:

        a=heapq.heappop(arr)
        b=heapq.heappop(arr)

        s=a+b

        cost+=s

        heapq.heappush(arr,s)

    return cost
```

# 3️⃣ Maximum Sum Combinations

🔗 [https://www.geeksforgeeks.org/problems/maximum-sum-combination](https://www.geeksforgeeks.org/problems/maximum-sum-combination)

Given

```
A = [1,4,2]
B = [3,5,1]
k = 4
```

Find **top k sums**.

---

### Key Idea

Largest sum starts from

```
A[n-1] + B[n-1]
```

Then explore neighbors.

Use:

```
max heap
visited set
```

---

### Python

```python
import heapq

def maxCombinations(A,B,K):

    A.sort()
    B.sort()

    n=len(A)

    heap=[]
    visited=set()

    heapq.heappush(heap,(-(A[n-1]+B[n-1]),n-1,n-1))
    visited.add((n-1,n-1))

    res=[]

    while K>0:

        s,i,j=heapq.heappop(heap)
        res.append(-s)

        if i-1>=0 and (i-1,j) not in visited:

            heapq.heappush(heap,(-(A[i-1]+B[j]),i-1,j))
            visited.add((i-1,j))

        if j-1>=0 and (i,j-1) not in visited:

            heapq.heappush(heap,(-(A[i]+B[j-1]),i,j-1))
            visited.add((i,j-1))

        K-=1

    return res
```

# 4️⃣ Choose and Swap

🔗 [https://www.geeksforgeeks.org/problems/choose-and-swap](https://www.geeksforgeeks.org/problems/choose-and-swap)

Example

```
ccad
```

Swap to get **lexicographically smallest string**.

---

### Greedy Idea

Find first position where a **smaller character appears later**.

Swap globally.

---

### Python

```python
class Solution:

    def chooseandswap(self, A):

        s=list(A)
        seen=set()

        for i,c in enumerate(s):

            for ch in map(chr,range(ord('a'),ord(c))):

                if ch in s[i+1:]:

                    for j in range(len(s)):

                        if s[j]==c:
                            s[j]=ch

                        elif s[j]==ch:
                            s[j]=c

                    return "".join(s)

        return A
```

# 5️⃣ Minimum Platforms

🔗 [https://www.geeksforgeeks.org/problems/minimum-platforms](https://www.geeksforgeeks.org/problems/minimum-platforms)

Given train

```
arrival[]
departure[]
```

Find **minimum platforms required**.

---

### Greedy Logic

Sort both arrays.

Move two pointers.

---

### Visual

```
arrival : 900 940 950
depart  : 910 1200 1120
```

Track:

```
platforms needed
```

---

### Python

```python
def findPlatform(arr,dep,n):

    arr.sort()
    dep.sort()

    i=j=0

    plat=0
    maxPlat=0

    while i<n and j<n:

        if arr[i] <= dep[j]:

            plat+=1
            maxPlat=max(maxPlat,plat)

            i+=1

        else:

            plat-=1
            j+=1

    return maxPlat
```
